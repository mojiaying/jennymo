# 字面量创建正则表达式

# 字面量创建不能识别变量

let str = 'aaaabbbb'
/u/.test(str) // false
/aa/.test(str) // true
let x = 'aa'
/x/.test(str) // false不能识别变量
eval(`/${x}/`).test(str) // true eval()把字符串变成JS表达式
