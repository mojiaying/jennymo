# switch

  1. 如果表达式等于 case 中的值，将执行此 case 代码段
  2. break 关键字会终止 switch 的执行
  3. 没有任何 case匹配时将执行default 代码块
  4. 如果case执行后缺少break则接着执行后面的语句 没有break
let error = 'warning';
switch (error) {
  case 'notice':
  case 'warning':
      console.log('警告或提示信息');
      break;
  case 'error':
      console.log('错误信息');
}
// 缺少break会继续执行
switch (1) {
  case 1:
    console.log(1);
  case 2:
    console.log(2);
  default:
    console.log("default");
}

使用表达式写法

function message(age) {
  switch (true) {
    case age < 15:
      console.log("儿童");
      break;
    case age < 25:
      console.log("青少年");
      break;
    case age < 40:
      console.log("青年");
      break;
    case age < 60:
      console.log("中年");
      break;
    case age < 100:
      console.log("老年");
      break;
    default:
      console.log("年龄输出错误");
  }
}
message(10);
