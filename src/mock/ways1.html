在mock/index.js 中遍历接口列表 拦截请求并返回接口数据

// 引入mockjs
import Mock from 'mockjs'
import dataBase from './data'
import pathMap from '../module/pathMap'
function response (res) {
  // res是一个请求对象，包含: url, type, body
  const lastIndex = res.url.lastIndexOf('/')
  return {
    code: 200,
    data: dataBase[res.url.slice(lastIndex+1)] || '没有定义Mock数据',
    message: '请求成功'
  }
}

// Mock函数
const { mock } = Mock

// 设置延时
Mock.setup({
  timeout: 400
})

Object.values(pathMap).forEach(res => {
  // 使用拦截规则拦截命中的请求，mock(url, post/get, 返回的数据);
  mock(res[0], res[1] || 'post', response)
})

axios 根据接口列表对象进行简单封装

import Axios from "axios"
import pathMap from './pathMap'
const env = process.env.NODE_ENV
// https地址检验
const urlPattern = /^(https?:\/\/|\/\/)?(localhost|(([\w-]+)\.)+([-a-z]+)|((25[0-5]|2[0-4]\d|1\d{2}|\d{1,2})\.){3}(25[0-5]|2[0-4]\d|1\d{2}|\d{1,2}))(:[1-9]\d{0,5})?\/?/i
export default function (pathName='', params, conf) {
  let sendUrl = ''
  let method = ''
  if(urlPattern.test(pathName.trim())){
    sendUrl = pathName
    method = 'get'
  } else {
    // 获取请求配置信息
    const mapArr = pathMap[pathName]
    sendUrl = mapArr[0]
    method = (mapArr[1] || 'post').toLowerCase()
  }
  const defConf = {
    url: sendUrl,
    method: method,
    responseType: 'json',
    emulateJSON: true,
    baseURL: '/',
    interFaceErrMsg: false, // 是否抛出后台实际错误信息
    headers: {
        'X-Requested-With': 'XMLHttpRequest',
        'Content-Type': 'application/x-www-form-urlencoded'
    },
    timeout: env !== 'production' ? 8000 : 36000 // 超时时间
  }
  let mergeConf = Object.assign(defConf, conf)
  if(mergeConf === 'get') {
    mergeConf.params = params || {}
  } else {
     mergeConf.data = params || {}
  }
  return Axios.request(mergeConf)

}
Axios.interceptors.request.use(conf => {
  return conf

})
Axios.interceptors.response.use(res => {
  return res
})
